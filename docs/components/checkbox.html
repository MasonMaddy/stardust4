Checkbox

Published v1.0.0 Updated 2026-06-03 DRI: Mason

Control-only checkbox for independent multi-select and boolean toggles. Three types — checked, unchecked, and intermediate — across six interaction states inside a fixed 44×44px hit area. Labels are composed in parent layouts, not in this component.

Label composition

This component has no hasLabel variant. Wrap control + text in a parent <label> with gap: spacing/stack-gap/default (8px). Helper and error text sit beside the label, linked via aria-describedby.

interface CheckboxProps {
  /** Controlled checked state */
  modelValue?:       boolean;
  /** Partial / select-all state — clears on first user toggle */
  indeterminate?:    boolean;
  /** Disables interaction */
  disabled?:         boolean;
  /** Validation / form error state */
  error?:            boolean;
  /** Visible label (or use external <label> with :id association) */
  label?:            string;
  /** Accessible name when no visible label is available */
  ariaLabel?:        string;
  /** Links an external error message via aria-describedby */
  errorDescribedBy?: string;
  id?:               string;
  name?:             string;
  value?:            string;
}

<script setup lang="ts">
import { ref, computed, watch, onMounted } from 'vue';

const props = withDefaults(defineProps<{
  modelValue?:       boolean;
  indeterminate?:    boolean;
  disabled?:         boolean;
  error?:            boolean;
  ariaLabel?:        string;
  errorDescribedBy?: string;
  id?:               string;
  name?:             string;
  value?:            string;
}>(), { modelValue: false, indeterminate: false, disabled: false, error: false });

const emit = defineEmits<{
  'update:modelValue':    [value: boolean];
  'update:indeterminate': [value: boolean];
}>();

const inputRef = ref<HTMLInputElement | null>(null);

const type = computed(() => {
  if (props.indeterminate) return 'intermediate';
  return props.modelValue ? 'checked' : 'unchecked';
});

function onChange(e: Event) {
  const input = e.target as HTMLInputElement;
  if (props.indeterminate) emit('update:indeterminate', false); // first click clears
  emit('update:modelValue', input.checked);
}

// Keep native indeterminate property in sync with prop
watch(() => props.indeterminate, (val) => {
  if (inputRef.value) inputRef.value.indeterminate = val ?? false;
}, { immediate: true });

onMounted(() => {
  if (inputRef.value && props.indeterminate) inputRef.value.indeterminate = true;
});
</script>

import SwiftUI

struct StardustCheckbox: View {
    @Binding var isChecked:       Bool
    @Binding var isIndeterminate: Bool
    var isDisabled: Bool   = false
    var hasError:   Bool   = false
    var label:      String? = nil

    // Stardust token colours
    private let primary        = Color(red: 0,     green: 0.467, blue: 0.420) // colour/action/primary
    private let errorRed       = Color(red: 0.839, green: 0.153, blue: 0.290) // colour/feedback/error/default
    private let disabledFill   = Color(red: 0.741, green: 0.741, blue: 0.741) // colour/action/disabled
    private let successSubtle  = Color(red: 0.875, green: 0.949, blue: 0.945) // colour/feedback/success/subtle
    private let successDefault = Color(red: 0.004, green: 0.639, blue: 0.616) // colour/feedback/success/default
    private let cyanSurface    = Color(red: 0.875, green: 0.969, blue: 0.969) // colour/surface/cyan
    private let borderStrong   = Color(red: 0.502, green: 0.502, blue: 0.502) // colour/border/strong
    private let textPrimary    = Color(red: 0.145, green: 0.180, blue: 0.204) // colour/text/text-primary

    @State private var isFocused = false
    @State private var isHovered = false
    @State private var isPressed = false

    private var boxFill: Color {
        if isDisabled { return (isChecked || isIndeterminate) ? disabledFill : .clear }
        if hasError   { return (isChecked || isIndeterminate) ? errorRed     : .clear }
        return (isChecked || isIndeterminate) ? primary : .clear
    }
    private var boxBorderColor: Color {
        if isDisabled { return disabledFill }
        if hasError   { return errorRed }
        return (isHovered || isFocused) ? textPrimary : borderStrong
    }
    private var haloFill: Color {
        if isPressed { return (isChecked || isIndeterminate) ? successDefault.opacity(0.3) : textPrimary.opacity(0.15) }
        if isHovered { return (isChecked || isIndeterminate) ? cyanSurface : Color(red: 0.965, green: 0.965, blue: 0.965) }
        if isFocused { return successSubtle }
        return .clear
    }

    var body: some View {
        HStack(spacing: 8) {
            // 44×44 hit area
            ZStack {
                Circle()
                    .fill(haloFill)
                    .overlay(isFocused ? Circle().stroke(successDefault, lineWidth: 2) : nil)
                    .frame(width: 44, height: 44)

                // 18×18 box
                RoundedRectangle(cornerRadius: 4) // radius/sm
                    .fill(boxFill)
                    .overlay(RoundedRectangle(cornerRadius: 4).stroke(boxBorderColor, lineWidth: 2))
                    .frame(width: 18, height: 18)
                    .overlay(Group {
                        if isIndeterminate {
                            Capsule().fill(Color.white).frame(width: 8, height: 2)
                        } else if isChecked {
                            Path { p in
                                p.move(to:    CGPoint(x: 3,  y: 9))
                                p.addLine(to: CGPoint(x: 7,  y: 13))
                                p.addLine(to: CGPoint(x: 15, y: 5))
                            }
                            .stroke(Color.white,
                                    style: StrokeStyle(lineWidth: 2, lineCap: .round, lineJoin: .round))
                            .frame(width: 18, height: 18)
                        }
                    })
            }
            .onTapGesture {
                guard !isDisabled else { return }
                if isIndeterminate { isIndeterminate = false }
                isChecked.toggle()
            }
            .onHover { isHovered = $0 }
            .simultaneousGesture(DragGesture(minimumDistance: 0)
                .onChanged { _ in isPressed = true }
                .onEnded   { _ in isPressed = false })

            if let text = label {
                Text(text)
                    .foregroundColor(isDisabled ? disabledFill : textPrimary)
            }
        }
        .accessibilityElement(children: .ignore)
        .accessibilityLabel(label ?? "")
        .accessibilityValue(isIndeterminate ? "mixed" : (isChecked ? "checked" : "unchecked"))
        .accessibilityAddTraits(.isButton)
        .disabled(isDisabled)
    }
}

import androidx.compose.foundation.border
import androidx.compose.foundation.clickable
import androidx.compose.foundation.interaction.*
import androidx.compose.foundation.layout.*
import androidx.compose.foundation.shape.CircleShape
import androidx.compose.foundation.shape.RoundedCornerShape
import androidx.compose.material3.Text
import androidx.compose.runtime.*
import androidx.compose.ui.Alignment
import androidx.compose.ui.Modifier
import androidx.compose.ui.draw.clip
import androidx.compose.ui.draw.drawBehind
import androidx.compose.ui.geometry.CornerRadius
import androidx.compose.ui.graphics.Color
import androidx.compose.ui.graphics.Path

// Stardust token colours
private val PrimaryGreen   = Color(0xFF00776B) // colour/action/primary
private val ErrorRed       = Color(0xFFD62749) // colour/feedback/error/default
private val DisabledFill   = Color(0xFFBDBDBD) // colour/action/disabled
private val SuccessSubtle  = Color(0xFFDFF2F1) // colour/feedback/success/subtle
private val SuccessDefault = Color(0xFF01A39D) // colour/feedback/success/default
private val BorderStrong   = Color(0xFF808080) // colour/border/strong
private val TextPrimary    = Color(0xFF252525) // colour/text/text-primary
private val CyanSurface    = Color(0xFFDFF8F8) // colour/surface/cyan
private val SurfaceGrey    = Color(0xFFF6F6F6) // colour/surface/grey

@Composable
fun StardustCheckbox(
    checked:         Boolean,
    onCheckedChange: (Boolean) -> Unit,
    modifier:        Modifier = Modifier,
    indeterminate:   Boolean  = false,
    disabled:        Boolean  = false,
    error:           Boolean  = false,
    label:           String?  = null,
) {
    val interactionSource = remember { MutableInteractionSource() }
    val isHovered by interactionSource.collectIsHoveredAsState()
    val isFocused by interactionSource.collectIsFocusedAsState()
    val isPressed by interactionSource.collectIsPressedAsState()

    val haloColor = when {
        isFocused -> SuccessSubtle
        isPressed -> if (checked || indeterminate) SuccessDefault.copy(alpha = 0.3f)
                     else TextPrimary.copy(alpha = 0.15f)
        isHovered -> if (checked || indeterminate) CyanSurface else SurfaceGrey
        else      -> Color.Transparent
    }
    val haloBorderColor = if (isFocused) SuccessDefault else Color.Transparent

    val boxFill = when {
        disabled                            -> if (checked || indeterminate) DisabledFill else Color.White
        error && (checked || indeterminate) -> ErrorRed
        checked || indeterminate            -> PrimaryGreen
        else                               -> Color.White
    }
    val boxBorderColor = when {
        disabled                   -> DisabledFill
        error                      -> ErrorRed
        isHovered || isFocused     -> TextPrimary
        checked || indeterminate   -> boxFill
        else                       -> BorderStrong
    }

    Row(
        verticalAlignment = Alignment.CenterVertically,
        modifier = modifier
            .clickable(
                interactionSource = interactionSource,
                indication        = null,
                enabled           = !disabled,
            ) {
                onCheckedChange(!checked)
            }
            .semantics {
                role             = Role.Checkbox
                this.contentDescription = label ?: ""
                stateDescription = when {
                    indeterminate -> "mixed"
                    checked       -> "checked"
                    else          -> "unchecked"
                }
            }
    ) {
        // 44dp hit area + halo
        Box(
            contentAlignment = Alignment.Center,
            modifier = Modifier
                .size(44.dp)
                .clip(CircleShape)
                .drawBehind {
                    drawCircle(haloColor)
                    if (haloBorderColor != Color.Transparent) {
                        drawCircle(haloBorderColor, style = Stroke(2.dp.toPx()))
                    }
                }
        ) {
            // 18dp visual box
            Box(
                contentAlignment = Alignment.Center,
                modifier = Modifier
                    .size(18.dp)
                    .drawBehind {
                        val r = CornerRadius(4.dp.toPx())
                        drawRoundRect(boxFill, cornerRadius = r)
                        drawRoundRect(boxBorderColor, cornerRadius = r,
                            style = Stroke(width = 2.dp.toPx()))
                    }
            ) {
                when {
                    indeterminate -> Box(
                        Modifier
                            .size(8.dp, 2.dp)
                            .background(Color.White, RoundedCornerShape(1.dp))
                    )
                    checked -> androidx.compose.foundation.Canvas(Modifier.size(12.dp)) {
                        val path = Path().apply {
                            moveTo(2.dp.toPx(),  6.dp.toPx())
                            lineTo(5.dp.toPx(),  9.dp.toPx())
                            lineTo(10.dp.toPx(), 3.dp.toPx())
                        }
                        drawPath(path, Color.White,
                            style = Stroke(2.dp.toPx(), cap = StrokeCap.Round, join = StrokeJoin.Round))
                    }
                }
            }
        }

        label?.let {
            Spacer(Modifier.width(8.dp))
            Text(it, color = if (disabled) DisabledFill else TextPrimary, fontSize = 16.sp)
        }
    }
}

Acceptance criteria

  • All 18 Figma variants (3 types × 6 states) render correctly on web, iOS, and Android
  • 44×44px / 44pt / 48dp hit area on every variant; 18px / 18pt / 18dp visual box centred
  • Hover, focus, pressed, disabled, and error states match token matrix in Section 4
  • Focus ring is circular halo (radius/full) — not Button's inner-border glow
  • Focus ring appears on :focus-visible / keyboard navigation only, never on mouse click
  • Native <input type="checkbox"> always present (visually hidden, not removed)
  • indeterminate sets aria-checked="mixed"; first user click clears it and sets checked
  • Error state sets aria-invalid="true" and links to error message via aria-describedby
  • prefers-reduced-motion disables all transition animations
  • Component ships without a label — external <label> / aria-labelledby required

7Open Questions

Decisions worth confirming before engineering finalises implementation.

⚠️
1 · Figma "intermediate" vs engineering "indeterminate" The Figma variant is named intermediate; HTML, ARIA, and all platform APIs use indeterminate. Confirm the canonical name for the prop and Figma layer before engineering ships, to avoid divergence in documentation.
⚠️
2 · Unchecked focus halo colour The spec shows unchecked focus halo as surface/grey + action/pressed border, but checked/intermediate focus uses feedback/success/subtle + action/pressed border. Confirm whether the unchecked halo fill intentionally differs (grey vs cyan-tinted).
ℹ️
3 · Label composition ownership Labels are deliberately out of scope for this component. Confirm whether a CheckboxField wrapper (checkbox + label + optional helper/error text) is a separate component on the roadmap, or whether teams compose it ad-hoc in their own layouts.
