Component

Pill

WIP v1.0.0 Updated 2026-06-10

A non-interactive status and information label used to surface priority, status, or category metadata at a glance. Four colour variants across three visual types and three sizes. The pill body is purely presentational — for interactive scenarios (dismiss, filter, select) use the Interactive Pill component.

4Usage Guidelines

✓ Do

  • Surface status, priority, or category metadata at a glance
  • Keep label text short — 1 to 3 words maximum
  • Pair with descriptive text — never rely on colour alone
  • Use sm for dense lists and table rows
  • Use lg for card headers and prominent status display
  • Use solid for primary/high-emphasis status
  • Use minimal for subtle tagging in dense UI

✗ Don't

  • Make the pill tappable — use Interactive Pill for that
  • Use more than 3 words in the label
  • Rely on colour alone to convey meaning
  • Use as navigation, a filter, or a button
  • Place more than 2–3 pills on a single content item
  • Use a single pill to convey multiple unrelated concepts

7Motion & Interaction

ℹ️
No motion or interaction. The Pill component is fully non-interactive and has no animated states. If a pill appears or disappears as part of a list update, the parent view handles any transition. For interactive pills with hover, focus, dismiss, or selection behaviour, use the Interactive Pill component.

SwiftUI 
import SwiftUI

enum PillColor  { case green, orange, purple, grey }
enum PillType   { case solid, ghost, minimal }
enum PillSize   { case sm, md, lg }

struct Pill: View {
    var color:        PillColor = .purple
    var type:         PillType  = .solid
    var size:         PillSize  = .sm
    var text:         String
    var leadingIcon:  AnyView?  = nil
    var trailingIcon: AnyView?  = nil

    // Stardust token colours
    private let accentSecondary = Color(red: 0.502, green: 0.408, blue: 0.729) // colour/accent/secondary
    private let errorDefault    = Color(red: 0.824, green: 0.114, blue: 0.243) // colour/feedback/error/default
    private let actionPrimary   = Color(red: 0,     green: 0.467, blue: 0.420) // colour/action/primary
    private let surfaceInverse  = Color(red: 0.071, green: 0.071, blue: 0.071) // colour/surface/inverse
    private let surfaceOrange   = Color(red: 0.996, green: 0.949, blue: 0.878) // colour/surface/orange (new)
    private let surfaceCyan     = Color(red: 0.875, green: 0.949, blue: 0.945) // colour/surface/cyan
    private let surfacePurple   = Color(red: 0.953, green: 0.945, blue: 1.0)   // colour/surface/purple

    var body: some View {
        HStack(spacing: 4) {
            if let icon = leadingIcon {
                icon
                    .frame(width: iconSize, height: iconSize)
                    .foregroundColor(textColor)
                    .accessibilityHidden(true)
            }
            Text(text)
                .font(labelFont)
                .foregroundColor(textColor)
                .lineLimit(1)
            if let icon = trailingIcon {
                icon
                    .frame(width: iconSize, height: iconSize)
                    .foregroundColor(textColor)
                    .accessibilityHidden(true)
            }
        }
        .padding(.horizontal, hPadding)
        .padding(.vertical, 4)
        .frame(height: height)
        .background(backgroundColor)
        .clipShape(Capsule()) // radius/full
        .overlay(type == .ghost ? Capsule().stroke(borderColor, lineWidth: 1) : nil)
    }

    private var height:    CGFloat  { size == .sm ? 24 : size == .md ? 32 : 44 }
    private var hPadding:  CGFloat  { size == .sm ? 8  : 12 }
    private var iconSize:  CGFloat  { size == .lg ? 24 : 16 }
    private var labelFont: Font     { size == .lg ? .system(size: 14) : .system(size: 12) }

    private var textColor: Color {
        switch type {
        case .solid: return .white
        case .ghost, .minimal:
            switch color {
            case .purple: return accentSecondary
            case .orange: return errorDefault
            case .green:  return actionPrimary
            case .grey:   return Color(red: 0.145, green: 0.180, blue: 0.204) // text/text-primary
            }
        }
    }
    private var backgroundColor: Color {
        switch (color, type) {
        case (.purple, .solid):   return accentSecondary
        case (.orange, .solid):   return errorDefault
        case (.green,  .solid):   return actionPrimary
        case (.grey,   .solid):   return surfaceInverse
        case (.purple, .ghost),  (.purple, .minimal):  return type == .ghost ? .white : surfacePurple
        case (.orange, .ghost),  (.orange, .minimal):  return type == .ghost ? .white : surfaceOrange
        case (.green,  .ghost),  (.green,  .minimal):  return type == .ghost ? .white : surfaceCyan
        case (.grey,   .ghost),  (.grey,   .minimal):  return type == .ghost ? .white : Color(red: 0.965, green: 0.965, blue: 0.965)
        }
    }
    private var borderColor: Color {
        switch color {
        case .purple: return accentSecondary
        case .orange: return errorDefault
        case .green:  return actionPrimary
        case .grey:   return Color(red: 0.741, green: 0.741, blue: 0.741) // text/text-disabled
        }
    }
}

Jetpack Compose (Kotlin) 
import androidx.compose.foundation.background
import androidx.compose.foundation.border
import androidx.compose.foundation.layout.*
import androidx.compose.foundation.shape.CircleShape
import androidx.compose.material3.Text
import androidx.compose.runtime.Composable
import androidx.compose.ui.Alignment
import androidx.compose.ui.Modifier
import androidx.compose.ui.draw.clip
import androidx.compose.ui.graphics.Color
import androidx.compose.ui.unit.Dp
import androidx.compose.ui.unit.dp
import androidx.compose.ui.unit.sp

enum class PillColor  { Green, Orange, Purple, Grey }
enum class PillType   { Solid, Ghost, Minimal }
enum class PillSize   { Sm, Md, Lg }

// Stardust token colours
private val AccentSecondary = Color(0xFF8068BA) // colour/accent/secondary
private val ErrorDefault    = Color(0xFFD21D3E) // colour/feedback/error/default
private val ActionPrimary   = Color(0xFF00776B) // colour/action/primary
private val SurfaceInverse  = Color(0xFF121212) // colour/surface/inverse
private val SurfaceOrange   = Color(0xFFFEF2E0) // colour/surface/orange (new)
private val SurfaceCyan     = Color(0xFFDFF2F1) // colour/surface/cyan
private val SurfacePurple   = Color(0xFFF3F1FF) // colour/surface/purple
private val SurfaceGrey     = Color(0xFFF6F6F6) // colour/surface/grey
private val TextDisabled    = Color(0xFFBDBDBD) // colour/text/text-disabled
private val TextPrimary     = Color(0xFF252525) // colour/text/text-primary

@Composable
fun Pill(
    text:           String,
    modifier:       Modifier  = Modifier,
    color:          PillColor = PillColor.Purple,
    type:           PillType  = PillType.Solid,
    size:           PillSize  = PillSize.Sm,
    leadingIcon:    @Composable (() -> Unit)? = null,
    trailingIcon:   @Composable (() -> Unit)? = null,
) {
    val height:   Dp   = when (size) { PillSize.Sm -> 24.dp; PillSize.Md -> 32.dp; PillSize.Lg -> 44.dp }
    val hPadding: Dp   = if (size == PillSize.Sm) 8.dp else 12.dp
    val iconSize: Dp   = if (size == PillSize.Lg) 24.dp else 16.dp
    val fontSize        = if (size == PillSize.Lg) 14.sp else 12.sp

    val bg: Color = when (type) {
        PillType.Solid   -> when (color) { PillColor.Purple -> AccentSecondary; PillColor.Orange -> ErrorDefault; PillColor.Green -> ActionPrimary; PillColor.Grey -> SurfaceInverse }
        PillType.Ghost   -> Color.White
        PillType.Minimal -> when (color) { PillColor.Purple -> SurfacePurple; PillColor.Orange -> SurfaceOrange; PillColor.Green -> SurfaceCyan; PillColor.Grey -> SurfaceGrey }
    }
    val textColor: Color = when (type) {
        PillType.Solid -> Color.White
        else           -> when (color) { PillColor.Purple -> AccentSecondary; PillColor.Orange -> ErrorDefault; PillColor.Green -> ActionPrimary; PillColor.Grey -> TextPrimary }
    }
    val borderColor: Color = when (color) { PillColor.Purple -> AccentSecondary; PillColor.Orange -> ErrorDefault; PillColor.Green -> ActionPrimary; PillColor.Grey -> TextDisabled }

    Row(
        verticalAlignment = Alignment.CenterVertically,
        horizontalArrangement = Arrangement.spacedBy(4.dp),
        modifier = modifier
            .height(height)
            .clip(CircleShape) // radius/full
            .background(bg)
            .then(if (type == PillType.Ghost) Modifier.border(1.dp, borderColor, CircleShape) else Modifier)
            .padding(horizontal = hPadding, vertical = 4.dp),
    ) {
        leadingIcon?.let {
            Box(Modifier.size(iconSize)) {
                CompositionLocalProvider(LocalContentColor provides textColor) { it() }
            }
        }
        Text(text, color = textColor, fontSize = fontSize, maxLines = 1)
        trailingIcon?.let {
            Box(Modifier.size(iconSize)) {
                CompositionLocalProvider(LocalContentColor provides textColor) { it() }
            }
        }
    }
}

Acceptance criteria

  • All 36 variants (4 colour × 3 type × 3 size) render correctly on web, iOS, and Android
  • Heights: sm = 24px, md = 32px, lg = 44px
  • radius/full applied — pill shape on all variants
  • All colour × type token combinations match the spec tables in Section 5
  • colour/surface/orange token exists in tokens.css
  • Leading icon and trailing flag at 16px (sm/md) and 24px (lg)
  • Both icons carry aria-hidden="true" or equivalent — decorative only
  • Pill body renders as <span> / non-interactive view — never <button> or <a>
  • No hover, focus, pressed, or active styles applied
  • Label text colour contrast verified against WCAG 1.4.3 for all 36 variants
  • Icons inherit text colour via currentColor / LocalContentColor
