Radio Button

WIP v1.0.0 Updated 2026-06-05

Lets users select exactly one option from a visible set of mutually exclusive choices. Control-only — labels are composed in the parent layout. Five current states; hover and pressed deferred to Phase 4 (web).

4Usage Guidelines

✓ Do

  • Use when choices are mutually exclusive and all options should remain visible
  • Settings pages, payment method, shipping speed, sort order
  • Always use in groups of two or more options
  • Pre-select a sensible default where one exists
  • Wrap the group in <fieldset> + <legend>

✗ Don't

  • Use when multiple selections are allowed → Checkbox
  • Use with more than ~6–8 options → Select / Dropdown
  • Use for immediate actions → Button
  • Use a single standalone radio button
  • Allow users to deselect all options by re-clicking

import SwiftUI

struct StardustRadioButton: View {
    @Binding var selectedValue: String
    let value:    String
    let label:    String
    var disabled: Bool = false

    // Stardust token colours
    private let primary        = Color(red: 0,     green: 0.467, blue: 0.420) // colour/action/primary
    private let borderStrong   = Color(red: 0.502, green: 0.502, blue: 0.502) // colour/border/strong
    private let disabledColor  = Color(red: 0.741, green: 0.741, blue: 0.741) // colour/text/text-disabled
    private let focusPrimary   = Color(red: 0.482, green: 0.792, blue: 0.773) // colour/focus/primary
    private let focusSecondary = Color(red: 0.741, green: 0.741, blue: 0.741) // colour/focus/secondary

    @State private var isHovered = false
    @State private var isPressed = false
    private var isSelected: Bool { selectedValue == value }

    var body: some View {
        Button(action: { if !disabled { selectedValue = value } }) {
            HStack(spacing: 8) {
                // 44pt touch target with halo
                ZStack {
                    Circle()
                        .fill(haloColor)
                        .frame(width: 44, height: 44)
                    // Glyph
                    ZStack {
                        Circle()
                            .stroke(isSelected ? primary : (disabled ? disabledColor : borderStrong), lineWidth: 2)
                            .frame(width: 18, height: 18)
                        if isSelected {
                            Circle()
                                .fill(disabled ? disabledColor : primary)
                                .frame(width: 9, height: 9)
                                .transition(.scale)
                        }
                    }
                    .scaleEffect(isPressed ? 0.85 : 1.0)
                    .animation(.easeIn(duration: 0.075), value: isPressed)
                }
                Text(label)
                    .foregroundColor(disabled ? disabledColor : .primary)
            }
        }
        .buttonStyle(.plain)
        .disabled(disabled)
        .onHover { isHovered = $0 }
        .simultaneousGesture(DragGesture(minimumDistance: 0)
            .onChanged { _ in isPressed = true }
            .onEnded   { _ in isPressed = false })
        .accessibilityLabel(label)
        .accessibilityAddTraits(isSelected ? [.isSelected] : [])
    }

    private var haloColor: Color {
        if isPressed { return isSelected ? Color(red: 0.004, green: 0.639, blue: 0.616).opacity(0.3) : Color.gray.opacity(0.15) }
        if isHovered { return isSelected ? Color(red: 0.875, green: 0.969, blue: 0.969) : Color(red: 0.965, green: 0.965, blue: 0.965) }
        return .clear
    }
}

Jetpack Compose (Kotlin) 
import androidx.compose.animation.core.*
import androidx.compose.foundation.clickable
import androidx.compose.foundation.interaction.*
import androidx.compose.foundation.layout.*
import androidx.compose.foundation.selection.selectableGroup
import androidx.compose.material3.Text
import androidx.compose.runtime.*
import androidx.compose.ui.Alignment
import androidx.compose.ui.Modifier
import androidx.compose.ui.draw.drawBehind
import androidx.compose.ui.graphics.Color
import androidx.compose.ui.graphics.drawscope.Stroke
import androidx.compose.ui.semantics.*
import androidx.compose.ui.unit.dp
import androidx.compose.ui.unit.sp

// Stardust token colours
private val PrimaryGreen   = Color(0xFF00776B) // colour/action/primary
private val BorderStrong   = Color(0xFF808080) // colour/border/strong
private val DisabledColor  = Color(0xFFBDBDBD) // colour/text/text-disabled
private val FocusPrimary   = Color(0xFF7BCAC5) // colour/focus/primary
private val FocusSecondary = Color(0xFFBDBDBD) // colour/focus/secondary
private val CyanSurface    = Color(0xFFDFF2F1) // colour/surface/cyan
private val GreySurface    = Color(0xFFF6F6F6) // colour/surface/grey
private val SuccessDefault = Color(0xFF01A39D) // colour/feedback/success/default
private val TextSecondary  = Color(0xFF838383) // colour/text/text-secondary

@Composable
fun StardustRadioButton(
    selected:   Boolean,
    onClick:    () -> Unit,
    modifier:   Modifier = Modifier,
    disabled:   Boolean  = false,
    label:      String?  = null,
) {
    val interactionSource = remember { MutableInteractionSource() }
    val isHovered  by interactionSource.collectIsHoveredAsState()
    val isFocused  by interactionSource.collectIsFocusedAsState()
    val isPressed  by interactionSource.collectIsPressedAsState()

    // Dot scale animation — 100ms Standard in, 100ms Standard Accelerate out
    val dotScale by animateFloatAsState(
        targetValue = if (selected) 1f else 0f,
        animationSpec = tween(
            durationMillis = 100,
            easing = if (selected) FastOutSlowInEasing else FastOutLinearInEasing
        ),
        label = "dotScale"
    )

    val haloColor = when {
        isFocused  -> if (selected) FocusPrimary   else FocusSecondary
        isPressed  -> if (selected) SuccessDefault else TextSecondary
        isHovered  -> if (selected) CyanSurface    else GreySurface
        else       -> Color.Transparent
    }
    val ringColor = if (disabled) DisabledColor else if (selected) PrimaryGreen else BorderStrong

    Row(
        verticalAlignment = Alignment.CenterVertically,
        modifier = modifier
            .clickable(
                interactionSource = interactionSource,
                indication        = null,
                enabled           = !disabled,
                onClick           = onClick,
            )
            .semantics {
                role             = Role.RadioButton
                this.selected    = selected
            }
    ) {
        // 44dp touch target with halo
        Box(
            contentAlignment = Alignment.Center,
            modifier = Modifier
                .size(44.dp)
                .drawBehind { drawCircle(haloColor) }
        ) {
            // Glyph — ring + dot
            androidx.compose.foundation.Canvas(Modifier.size(20.dp)) {
                // Ring
                drawCircle(ringColor, radius = 8.dp.toPx(), style = Stroke(2.dp.toPx()))
                // Inner dot (scale animated)
                if (dotScale > 0f) {
                    drawCircle(
                        if (disabled) DisabledColor else PrimaryGreen,
                        radius = 4.5.dp.toPx() * dotScale,
                    )
                }
            }
        }

        label?.let {
            Spacer(Modifier.width(8.dp))
            Text(it, color = if (disabled) DisabledColor else Color.Unspecified, fontSize = 16.sp)
        }
    }
}

// Usage — wrap in Column with selectableGroup()
@Composable
fun RadioGroup(
    options:        List<Pair<String, String>>, // value to label
    selectedValue:  String,
    onSelectionChange: (String) -> Unit,
    groupLabel:     String,
) {
    Column(Modifier.selectableGroup().semantics { contentDescription = groupLabel }) {
        options.forEach { (value, label) ->
            StardustRadioButton(
                selected = selectedValue == value,
                onClick  = { onSelectionChange(value) },
                label    = label,
            )
        }
    }
}

Acceptance criteria

  • All 5 current Figma states render correctly on web, iOS, and Android
  • 44×44px / 44pt / 48dp touch target; 20px / 18pt / 18dp glyph centred
  • Focus halo: colour/focus/secondary for focused, colour/focus/primary for selected-focused
  • Focus ring on :focus-visible / keyboard navigation only — never on mouse click
  • Dot scales 0 → 1 at 100ms Standard on selection; 1 → 0 at 100ms Standard Accelerate on deselection
  • State layer enter 200ms Standard, exit 150ms Standard Accelerate — scales from scale(0)
  • Press feedback: glyph scale(0.85) at 75ms Standard on press
  • Roving tabindex: Arrow keys move focus AND select; Tab enters/exits group; Enter does nothing
  • Clicking a selected radio has no effect
  • Group: <fieldset> + <legend> or role="radiogroup" + aria-labelledby
  • Native <input type="radio"> — visually hidden, not removed
  • Disabled: no pointer events, aria-disabled="true", grey ring via token
  • prefers-reduced-motion: all transitions disabled, state changes instant
  • Phase 4: hover/selected-hover/pressed/selected-pressed variants added to Figma + implemented
  • Phase 4: glyph as inline SVG with live colour/action/primary + colour/border/strong token bindings
